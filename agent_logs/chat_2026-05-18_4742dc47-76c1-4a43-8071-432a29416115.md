## 📋 Što je implementirano:

### 1. **HTML Struktura**
- Dva vidljiva input polja (`form-control` class) za datum i vrijeme
- Dinamički generirane ID-eve na osnovu `fieldNameDate` parametra
  - `fieldNameDate_date` za datum
  - `fieldNameDate_time` za vrijeme
- Skriveni input polje za submission: `fieldNameDate`

### 2. **Format Podrška**
- **hr-HR**: `dd.MM.yyyy HH:mm` (automatska detekcija iz `navigator.language`)
- **en-US**: `MM/dd/yyyy hh:mm tt`
- Auto-detektiranje formata iz browser kulture

### 3. **jQuery Validacija & Auto-formatting**
- Real-time formatting dok korisnik tipka
- Automatski dodavanje separatora (`.` ili `/`)
- Validacija regex patternima
- Prikaz/skrivanje error stanja

### 4. **CSS Stilovi s Varijablama**
- `--bg-surface` - pozadina
- `--accent-red` - focus boja
- `--text-primary` - primarni tekst
- `--text-muted` - natuškana boja
- `--win` - zelena (validni input)
- `--loss` - crvena (invalid input)

### 5. **Funkcionalnosti**
- ✅ Error/valid state sa vizualnom povratnom informacijom
- ✅ Form submission validacija
- ✅ Hidden input kombinira datum i vrijeme
- ✅ Responsive grid (side-by-side na desktopu, stack na mobilnom)

### 6. **JavaScript API** (window.DateTimePicker)
```javascript
DateTimePicker.getValue(fieldName)      // Dohvata kombiniranu vrijednost
DateTimePicker.setValue(fieldName, date, time)  // Postavlja vrijednosti
DateTimePicker.clear(fieldName)         // Briše sve
DateTimePicker.validate(fieldName)      // Validira
```

    ? /^\d{2}\.\d{2}\.\d{4}$/ // dd.MM.yyyy
    : /^\d{2}\/\d{2}\/\d{4}$/; // MM/dd/yyyy

    var val = $(this).val().replace(/\D/g, ''); // Ukloni ne-brojeve

        // dd.MM.yyyy - ubaci točke nakon svakih 2 cifre

        if (val.length >= 5) val = val.slice(0,5) + '.' + val.slice(5,9);
    } else {
        // MM/dd/yyyy
        if (val.length >= 2) val = val.slice(0,2) + '/' + val.slice(2);
        if (val.length >= 5) val = val.slice(0,5) + '/' + val.slice(5,9);
    }
    
    $(this).val(val);
    validateDate();
});

// AUTO-FORMATIRANJE VREMENA
$timeInput.on('input', function() {
    var val = $(this).val().replace(/\D/g, ''); // Ukloni ne-brojeve
    
    // HH:mm - ubaci dvotačku nakon 2 cifre
    if (val.length >= 2) val = val.slice(0,2) + ':' + val.slice(2,4);
    
    $(this).val(val);
    validateTime();
});

// VALIDACIJA
function validateDate() {
    var val = $dateInput.val();
    if (!datePattern.test(val)) {
        $dateInput.addClass('invalid');
        $error.show();
    } else {
        $dateInput.removeClass('invalid');
        $error.hide();
    }
    updateHidden();
}

function validateTime() {
    var val = $timeInput.val();
    if (!timePattern.test(val)) {
        $timeInput.addClass('invalid');
    } else {
        $timeInput.removeClass('invalid');
    }
    updateHidden();
}

// KOMBINIRANJE DATUMA + VREMENA U SKRIVENI UNOS
function updateHidden() {
    var date = $dateInput.val(); // "15.05.2026"
    var time = $timeInput.val(); // "14:30"
    
    if (date && time) {
        $hidden.val(date + ' ' + time); // "15.05.2026 14:30"
    }
}

$timeInput.on('blur', validateTime);

Zamijenio sam C# kod u `<option>` atributima s `asp-selected` tag helperom:

**Prije:**
```html
<option value="@role" @(Model?.Role == role ? "selected" : "")>@role</option>
```

**Sada:**
```html
<option value="@role" asp-selected="Model?.Role == role">@role</option>
```

**Gdje:** 
- [Views/Player/Create.cshtml](Views/Player/Create.cshtml) linija 178

### 2. **CSS Escape Problem** - @keyframes i @media
Zamijenio sam `@` s `@@` da CSS znakovi budu interpretirani kao literalni tekst:

**Prije:**
```css
@keyframes spin { ... }
@media (max-width: 576px) { ... }
```

**Sada:**
```css
@@keyframes spin { ... }
@@media (max-width: 576px) { ... }
```

**Gdje:**
- [Views/Champion/Index.cshtml](Views/Champion/Index.cshtml) linija 192
- [Views/Shared/_DateTimePicker.cshtml](Views/Shared/_DateTimePicker.cshtml) linija 88

// Services/MatchHistoryBackgroundService.cs
using Microsoft.Extensions.Hosting;
using Microsoft.Extensions.DependencyInjection;
using PostMatchSummary.Services;

namespace PostMatchSummary.Services
{
    public class MatchHistoryBackgroundService : BackgroundService
    {
        private readonly IServiceProvider _serviceProvider;
        private readonly ILogger<MatchHistoryBackgroundService> _logger;
        private readonly TimeSpan _interval = TimeSpan.FromMinutes(20);

        public MatchHistoryBackgroundService(
            IServiceProvider serviceProvider,
            ILogger<MatchHistoryBackgroundService> logger)
        {
            _serviceProvider = serviceProvider;
            _logger = logger;
        }

        protected override async Task ExecuteAsync(CancellationToken stoppingToken)
        {
            // Čeka 1 minutu prije prvog pokretanja (daj vremena app-u da se pokrene)
            await Task.Delay(TimeSpan.FromMinutes(1), stoppingToken);

            while (!stoppingToken.IsCancellationRequested)
            {
                try
                {
                    _logger.LogInformation("🔄 Počinjem refresh match history-ja...");
                    
                    using (var scope = _serviceProvider.CreateScope())
                    {
                        var riotService = scope.ServiceProvider.GetRequiredService<RiotServices>();
                        var dbContext = scope.ServiceProvider.GetRequiredService<PostMatchSummaryDbContext>();

                        // Dohvati sve igrače iz baze
                        var players = await dbContext.Players
                            .Include(p => p.Champion)
                            .ToListAsync(stoppingToken);

                        foreach (var player in players)
                        {
                            try
                            {
                                _logger.LogInformation($"📊 Osvežavam match history za: {player.SummonerName}");
                                
                                // Pozovi Riot API za taj igrača
                                // (trebam metodu u RiotServices koja vraća match list)
                                var matchIds = await riotService.GetPlayerMatchHistoryAsync(
                                    player.SummonerName, 
                                    count: 5, 
                                    stoppingToken);

                                // Za svaki match, ako ne postoji, spremi u bazu
                                foreach (var matchId in matchIds)
                                {
                                    var existingMatch = await dbContext.Matches
                                        .FirstOrDefaultAsync(m => m.MatchId == matchId, stoppingToken);

                                    if (existingMatch == null)
                                    {
                                        // Dohvati detalje matcha
                                        var matchDetails = await riotService.GetMatchDetailsAsync(
                                            matchId, 
                                            stoppingToken);

                                        if (matchDetails != null)
                                        {
                                            dbContext.Matches.Add(matchDetails);
                                        }
                                    }
                                }

                                await dbContext.SaveChangesAsync(stoppingToken);
                            }
                            catch (Exception ex)
                            {
                                _logger.LogError($"❌ Greška pri osvežavanju {player.SummonerName}: {ex.Message}");
                            }
                        }

                        _logger.LogInformation("✅ Refresh završen!");
                    }
                }
                catch (Exception ex)
                {
                    _logger.LogError($"❌ Greška u background servisu: {ex.Message}");
                }

                // Čekaj 20 minuta prije sljedećeg pokretanja
                await Task.Delay(_interval, stoppingToken);
            }
        }
    }
}

// Services/RiotServices.cs - dodaj ove metode:

public async Task<List<string>> GetPlayerMatchHistoryAsync(
    string summonerName, 
    int count = 5, 
    CancellationToken cancellationToken = default)
{
    try
    {
        // 1. Prvo dohvati PUUID igrača (trebam Account API)
        var accountUrl = $"https://europe.api.riotgames.com/riot/account/v1/accounts/by-summoner-name/{summonerName}?api_key={_apiKey}";
        var response = await _httpClient.GetAsync(accountUrl, cancellationToken);
        
        if (!response.IsSuccessStatusCode) return new List<string>();

        var accountData = await response.Content.ReadAsAsync<dynamic>();
        string puuid = accountData.puuid;

        // 2. Dohvati match IDs za tog igrača
        var matchUrl = $"https://europe.api.riotgames.com/lol/match/v5/matches/by-puuid/{puuid}/ids?start=0&count={count}&api_key={_apiKey}";
        var matchResponse = await _httpClient.GetAsync(matchUrl, cancellationToken);
        
        if (!matchResponse.IsSuccessStatusCode) return new List<string>();

        var matchIds = await matchResponse.Content.ReadAsAsync<List<string>>();
        return matchIds ?? new List<string>();
    }
    catch (Exception ex)
    {
        _logger.LogError($"Greška pri dohvaćanju match history: {ex.Message}");
        return new List<string>();
    }
}

public async Task<Match?> GetMatchDetailsAsync(
    string matchId, 
    CancellationToken cancellationToken = default)
{
    try
    {
        var url = $"https://europe.api.riotgames.com/lol/match/v5/matches/{matchId}?api_key={_apiKey}";
        var response = await _httpClient.GetAsync(url, cancellationToken);
        
        if (!response.IsSuccessStatusCode) return null;

        var matchData = await response.Content.ReadAsAsync<dynamic>();
        
        // Mapira Riot API response na Match model
        var match = new Match
        {
            MatchId = matchId,
            GameCreation = DateTimeOffset.FromUnixTimeMilliseconds((long)matchData.info.gameCreation).DateTime,
            Duration = (int)matchData.info.gameDuration,
            GameMode = matchData.info.gameMode,
            GameVersion = matchData.info.gameVersion
        };

        return match;
    }
    catch (Exception ex)
    {
        _logger.LogError($"Greška pri dohvaćanju match detalja: {ex.Message}");
        return null;
    }
}
