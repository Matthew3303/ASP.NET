```csharp
public IActionResult Search(string? q)
{
    var query = _dbContext.Champions.OrderBy(c => c.Name).AsQueryable();

    if (!string.IsNullOrWhiteSpace(q))
    {
        var lower = q.ToLower();
```

```javascript
function renderDropdown(items) {
    dropdown.empty();
    
    if (items.length === 0) {
        dropdown.html('<div class="autocomplete-empty">No results found</div>').addClass('show');
        return;
    }

    $.each(items, function (i, item) {
        var div = $('<div class="autocomplete-item">')
            .text(item[displayField])  // Prikazuj ime
            .attr('data-id', item.id)
            .attr('data-label', item[displayField]);
        dropdown.append(div);
    });

    dropdown.addClass('show');
}

// Debounce za AJAX pretragu
displayInput.on('input', function (e) {
    clearTimeout(debounceTimer);
    var query = $(this).val();
    
    debounceTimer = setTimeout(function () {
        $.ajax({
            url: endpoint,
            method: 'GET',
            data: { q: query },  // 📤 Šalji pretragu
            success: function (data) {
                renderDropdown(data);  // 📥 Prikaži rezultate
            }
        });
    }, 300);  // Čekaj 300ms nakon zadnje tipke
});

// Keyboard navigacija
displayInput.on('keydown', function (e) {
    if (e.key === 'ArrowDown') {
        selectedIndex = Math.min(selectedIndex + 1, filteredItems.length - 1);
        highlightItem(selectedIndex);  // ⬇️ Ide na sljedeći item
    }
    else if (e.key === 'ArrowUp') {
        selectedIndex = Math.max(selectedIndex - 1, 0);
        highlightItem(selectedIndex);  // ⬆️ Ide na prethodni item
    }
    else if (e.key === 'Enter') {
        var item = filteredItems[selectedIndex];
        selectItem(item.id, item[displayField]);  // ✅ Odaberi
    }
});
```
**Objašnjenje:** 
- **Debounce** - čeka 300ms nakon što korisnik prestane pisati pa tek onda šalje AJAX
- **Arrow Down/Up** - navigira po stavkama
- **Enter** - odabira trenutnu stavku
- **Tab** - odabira prvi rezultat i zatvara dropdown

```javascript
function formatDate(val) {
    var d = val.replace(/\D/g, '');  // Uklanja sve što nije broj
    if (d.length >= 5) return d.slice(0,2) + '.' + d.slice(2,4) + '.' + d.slice(4,8);
    if (d.length >= 3) return d.slice(0,2) + '.' + d.slice(2,4);
    if (d.length >= 2) return d.slice(0,2) + '.';
    return d;
}

function formatTime(val) {
    var d = val.replace(/\D/g, '');
    if (d.length >= 3) return d.slice(0,2) + ':' + d.slice(2,4);
    if (d.length >= 2) return d.slice(0,2) + ':';
    return d;
}
```

```javascript
function renderDropdown(items) {
    dropdown.empty();          // Očisti stare rezultate
    selectedIndex = -1;        // Resetiraj selekciju

    // Ako nema rezultata
    if (items.length === 0) {
        dropdown.html('<div class="autocomplete-empty">No results found</div>').addClass('show');
        filteredItems = [];
        return;
    }

    filteredItems = items;  // Spremi za keyboard navigaciju

    // Kreiraj div za svaki rezultat
    $.each(items, function (i, item) {
        var div = $('<div class="autocomplete-item">')
            .text(item[displayField])      // Prikaži: "Garen"
            .attr('data-id', item.id)      // Spremi ID: 1
```

```javascript
        highlightItem(selectedIndex);
    }
    
    else if (e.key === 'Tab' || e.key === 'Enter') {  // ↵ Enter ili Tab
        if (isOpen && selectedIndex >= 0) {  // Ako je nešto selektovano
            e.preventDefault();
            var item = filteredItems[selectedIndex];
            selectItem(item.id, item[displayField]);
        } else if (isOpen && filteredItems.length > 0) {  // Ako postoji rezultat
            e.preventDefault();
            var first = filteredItems[0];
            selectItem(first.id, first[displayField]);
        }
    }
    
    else if (e.key === 'Escape') {  // Esc - zatvori
        dropdown.removeClass('show').html('');
        selectedIndex = -1;
    }
});

function highlightItem(index) {
    // Ukloni highlight sa svih stavki
    dropdown.find('.autocomplete-item').removeClass('selected');
    
    if (index >= 0 && index < filteredItems.length) {
        // Dodaj highlight na trenutnu
        var item = dropdown.find('.autocomplete-item[data-index="' + index + '"]');
        item.addClass('selected');
        item[0].scrollIntoView({ block: 'nearest' });  // Scrollaj u view
    }
}
```

```javascript
$(function () {
    var fieldName = '@fieldName';
    var endpoint = '@endpoint';
    var displayField = '@displayField';
    
    var displayInput = $('#' + fieldName + '_display');
    var dropdown = $('#' + fieldName + '_dropdown');
    var hiddenInput = $('#' + fieldName);
    var allItems = [];
    var filteredItems = [];
    var selectedIndex = -1;
    var debounceTimer;  // 🎯 Ključno za animaciju - delay prije slanja

    // 📍 RENDERIRANJE DROPDOWN STAVAKA
    function renderDropdown(items) {
        dropdown.empty();
        selectedIndex = -1;

        if (items.length === 0) {
            dropdown.html('<div class="autocomplete-empty">No results found</div>')
                   .addClass('show');  // Dodaj .show → display: block
            filteredItems = [];
            return;
        }

        filteredItems = items;

        // Kreiraj svaki item s podatcima
        $.each(items, function (i, item) {
            var div = $('<div class="autocomplete-item">')
                .text(item[displayField])
                .attr('data-id', item.id)
                .attr('data-label', item[displayField])
                .attr('data-index', i);
            dropdown.append(div);
        });

        dropdown.addClass('show');  // 📍 ANIMACIJA: Prikaži dropdown

        // Klikni na item
        dropdown.find('.autocomplete-item').on('click', function () {
            selectItem($(this).data('id'), $(this).data('label'));
        });
    }

    // 📍 HIGHLIGHT STAVKE - Keyboard navigacija
    function highlightItem(index) {
        dropdown.find('.autocomplete-item').removeClass('selected');
        if (index >= 0 && index < filteredItems.length) {
            var item = dropdown.find('.autocomplete-item[data-index="' + index + '"]');
            item.addClass('selected');  // Dodaj red background
            item[0].scrollIntoView({ block: 'nearest' });
        }
    }

    // 📍 ANIMACIJA: Keyboard navigacija - ArrowDown, ArrowUp
    displayInput.on('keydown', function (e) {
        var isOpen = dropdown.hasClass('show') && filteredItems.length > 0;

        if (e.key === 'ArrowDown') {
            e.preventDefault();
            if (!isOpen) {
                renderDropdown(allItems.length > 0 ? allItems : filteredItems);
                return;
            }
            // Premjesti highlight dolje
            selectedIndex = Math.min(selectedIndex + 1, filteredItems.length - 1);
            highlightItem(selectedIndex);  // 📍 Updateaj highlight sa CSS
        }
        else if (e.key === 'ArrowUp') {
            e.preventDefault();
            // Premjesti highlight gore
            selectedIndex = Math.max(selectedIndex - 1, 0);
            highlightItem(selectedIndex);
        }
        else if (e.key === 'Tab' || e.key === 'Enter') {
            // Selektiraj highlighted item
            if (isOpen && selectedIndex >= 0) {
                e.preventDefault();
                var item = filteredItems[selectedIndex];
                selectItem(item.id, item[displayField]);
            }
        }
        else if (e.key === 'Escape') {
            dropdown.removeClass('show').html('');
            selectedIndex = -1;
        }
    });

    // 📍 ANIMACIJA: DEBOUNCE - Delay prije slanja AJAX
    displayInput.on('input', function () {
        var query = $(this).val().trim();
        hiddenInput.val('');
        selectedIndex = -1;

        clearTimeout(debounceTimer);  // Obriši prethodni timer

        if (query.length === 0) {
            renderDropdown(allItems);
            return;
        }

        // 📍 KLJUČNA ANIMACIJA: Čekaj 200ms prije slanja zahtjeva
        debounceTimer = setTimeout(function () {
            // Prikaži loading stavku dok se čeka odgovor
            dropdown.html('<div class="autocomplete-loading">🔍 Searching...</div>')
                   .addClass('show');
            
            $.ajax({
                url: endpoint,
                method: 'GET',
                data: { q: query },
                success: function (data) {
                    renderDropdown(data);  // 📍 Prikaži rezultate s animation
                },
                error: function () {
                    dropdown.html('<div class="autocomplete-empty">Error searching</div>')
                           .addClass('show');
                }
            });
        }, 200);  // ← 200ms delay (DEBOUNCE)
    });

    // Focus event - prikaži sve stavke
    displayInput.on('focus', function () {
        if (allItems.length > 0) {
            renderDropdown(allItems);
        } else {
            loadAll();
        }
    });

    // Blur event - zatvori dropdown nakon 200ms
    displayInput.on('blur', function () {
        setTimeout(function () {
            dropdown.removeClass('show');  // 📍 Ukloni display
        }, 200);
    });
});
```

```javascript
    var dateInput = $('#@dateInputId');      /* ID: GameCreation_date */
    var timeInput = $('#@timeInputId');      /* ID: GameCreation_time */

    // 📍 FUNKCIJA 1: Auto-format datuma
    function formatDate(val) {
        // Ukloni sve znakove koji nisu brojevi
        var d = val.replace(/\D/g, '');
        
        // Primjer: "01012026" → "01.01.2026"
        if (d.length >= 5) {
            return d.slice(0,2) + '.' + d.slice(2,4) + '.' + d.slice(4,8);
        }
        // Primjer: "010" → "01.0"
        if (d.length >= 3) {
            return d.slice(0,2) + '.' + d.slice(2,4);
        }
        // Primjer: "01" → "01."
        if (d.length >= 2) {
            return d.slice(0,2) + '.';
        }
        // Primjer: "0" → "0"
        return d;
    }

    // 📍 FUNKCIJA 2: Auto-format vremena
    function formatTime(val) {
        // Ukloni sve znakove koji nisu brojevi
        var d = val.replace(/\D/g, '');
        
        // Primjer: "1430" → "14:30"
        if (d.length >= 3) {
            return d.slice(0,2) + ':' + d.slice(2,4);
        }
        // Primjer: "14" → "14:"
        if (d.length >= 2) {
            return d.slice(0,2) + ':';
        }
        // Primjer: "1" → "1"
        return d;
    }

    // 📍 FUNKCIJA 3: Validacija datuma - Regex
    function validateDate(val) {
        // Pattern: DD.MM.YYYY
        // Dozvoljeni dani: 01-31
        // Dozvoljeni mjeseci: 01-12
        // Dozvoljene godine: bilo koja 4-znamenkasta
        return /^(0[1-9]|[12]\d|3[01])\.(0[1-9]|1[0-2])\.\d{4}$/.test(val);
    }

    // 📍 FUNKCIJA 4: Validacija vremena - Regex
    function validateTime(val) {
        // Pattern: HH:MM (24-satni format)
        // Dozvoljeni sati: 00-23
        // Dozvoljene minute: 00-59
        return /^([01]\d|2[0-3]):[0-5]\d$/.test(val);
    }

    // 📍 FUNKCIJA 5: Kombiniraj datum i vrijeme u hidden input
    function updateHidden() {
        var d = dateInput.val();  /* "01.01.2026" */
        var t = timeInput.val();  /* "14:30" */
        
        if (d && t) {
            // Oba su unešena: "01.01.2026 14:30"
            hiddenInput.val(d + ' ' + t);
        }
        else if (d) {
            // Samo datum: "01.01.2026 00:00"
            hiddenInput.val(d + ' 00:00');
        }
        else {
            // Ništa nije unešeno: ""
            hiddenInput.val('');
        }
    }

    dateInput.on('input', function () {
        var pos = this.selectionStart;
        
        // 1. Formatiraj unesen tekst
        var formatted = formatDate($(this).val());
        $(this).val(formatted);
        
        // 2. Validira li se?
        var isValid = validateDate(formatted);
        
        // 3. Ukloni stare klase
        $(this).removeClass('valid invalid');

        if (formatted.length >= 8) {
            $(this).addClass(isValid ? 'valid' : 'invalid');
            // valid → zeleni okvir
            // invalid → crveni okvir

        updateHidden();

        if (formatted.length >= 4) {

            // valid → zeleni okvir
            // invalid → crveni okvir
        }
        
        // 5. Updateaj hidden input
        updateHidden();
    });

    // 📍 EVENT 3: Date blur - prikaži grešku ako nije validan
    dateInput.on('blur', function () {
        var errorDiv = $(this).closest('.datetime-picker-group')
                              .find('.datetime-picker-error');
        
        if ($(this).val() && !validateDate($(this).val())) {
            // Ako korisnik unese nešto, ali nije validno
            errorDiv.text('Format: DD.MM.YYYY').addClass('show');
        } else {
            // Ako je validno, ukloni grešku
            errorDiv.removeClass('show');
        }
    });

    // 📍 EVENT 4: Time blur - prikaži grešku ako nije validan
```

```javascript
// 📍 VALIDACIJA DATUMA - Sve kombinacije
function validateDate(val) {
    // Pattern objašnjenje:
    // (0[1-9]|[12]\d|3[01])  → Dani: 01-09, 10-29, 30-31
    // (0[1-9]|1[0-2])        → Mjeseci: 01-09, 10-12
    // \d{4}                  → Godina: bilo koja 4 znamenke
    
    var pattern = /^(0[1-9]|[12]\d|3[01])\.(0[1-9]|1[0-2])\.\d{4}$/;
    return pattern.test(val);

validateDate("32.01.2026")  → false ❌ (dan 32 ne postoji)
```
